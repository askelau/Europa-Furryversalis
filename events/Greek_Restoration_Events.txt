namespace = greek_restoration

country_event = {
	id = greek_restoration.1
	title = greek_restoration.1.t
	desc = greek_restoration.1.d
	picture = DIPLOMACY_eventPicture
	is_triggered_only = yes

	option = {
		name = greek_restoration.1.a
		151 = { move_capital_effect = yes }

		every_subject_country = {
			limit = {
				OR = {
					is_subject_of_type = pronoia_subject_type
					is_subject_of_type = hereditary_pronoia_subject_type
				}
			}
			add_liberty_desire = -100
		}

		add_country_modifier = {
			name = greek_reorganized_pronoias
			duration = 7300
		}

		change_tag = BYZ
		swap_non_generic_missions = yes
		on_change_tag_effect = yes

		if = {
			limit = {
				NOT = { culture_group = byzantine }
				NOT = { accepted_culture = greek }
			}
			add_accepted_culture = greek
		}
		if = {
			limit = { NOT = { government = monarchy } }
			change_government = monarchy
		}
		if = {
			limit = { has_not_locked_first_tier_reform = yes }
			add_government_reform = byzantine_autocracy_reform
		}

		set_government_rank = 3

		if = {
			limit = {
				has_custom_ideas = no
				NOT = { has_idea_group = BYZ_ideas }
			}
			country_event = { id = ideagroups.1 }
		}

		set_country_flag = formed_byzantium_flag
		add_prestige = 25
	}

	option = {
		name = greek_restoration.1.b

		every_subject_country = {
			limit = {
				OR = {
					is_subject_of_type = pronoia_subject_type
					is_subject_of_type = hereditary_pronoia_subject_type
				}
			}
			ROOT = { inherit = PREV }
		}

		change_tag = MCE
		if = {
			limit = { NOT = { government = monarchy } }
			change_government = monarchy
		}
		add_government_reform = alexandrian_empire_reform
		set_government_rank = 3
		swap_non_generic_missions = yes
		on_change_tag_effect = yes

		if = {
			limit = {
				has_custom_ideas = no
				NOT = { has_idea_group = MCE_ideas }
			}
			country_event = { id = ideagroups.1 }
		}

		add_prestige = 50
	}

	option = {
		name = greek_restoration.1.c
		set_government_rank = 3
		add_prestige = 25
	}
}
