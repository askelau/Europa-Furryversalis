		position = 2

		trigger = {
			thrace_area = {
				type = all

		}

		provinces_to_highlight = {
			area = thrace_area
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}

		effect = {
			macedonia_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }

				add_permanent_claim = ROOT
			}
			every_province = {
				limit = {
					OR = {
						area = northern_greece_area
						area = morea_area

					NOT = { province_id = 142 } # Corfu
					NOT = { province_id = 163 } # Crete
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
