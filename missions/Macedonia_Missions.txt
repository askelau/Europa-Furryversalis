macedonia_center_missions = {
	slot = 2
	generic = no
	ai = yes
	potential = {
		NOT = { map_setup = map_setup_random }
		tag = MCD
	}
	has_country_shield = yes

	mcd_gain_independence = {
		icon = mike_mission
		required_missions = { }
		position = 1
		trigger = { is_subject = no }
		effect = {
			thrace_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			add_prestige = 10
		}
	}

	mcd_conquer_thrace = {
		icon = mission_conquer_50_development
		required_missions = { mcd_gain_independence }
		position = 2
		trigger = {
			thrace_area = {
				type = all
				country_or_non_sovereign_subject_holds = ROOT
			}
		}
		provinces_to_highlight = {
			area = thrace_area
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		effect = {
			macedonia_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			every_province = {
				limit = {
					OR = {
						area = northern_greece_area
						area = morea_area
					}
					NOT = { province_id = 142 } # Corfu
					NOT = { province_id = 163 } # Crete
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			hudavendigar_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			aydin_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			germiyan_area = {
				limit = {
					NOT = { is_permanent_claim = ROOT }
					NOT = { is_core = ROOT }
				}
				add_permanent_claim = ROOT
			}
			add_country_modifier = {
				name = greek_thracian_reconquest
				duration = 5475
			}
		}
	}

	mcd_restore_empire = {
		icon = mission_empire
		required_missions = {
			mcd_secure_balkans
			mcd_conquer_anatolia
		}
		position = 5
		trigger = {
			num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
				value = 40
				culture_group = byzantine
				OR = {
					region = balkan_region
					region = anatolia_region
					region = caucasia_region
				}
			}
		}
		effect = {
			country_event = { id = greek_restoration.1 }
		}
	}
}
